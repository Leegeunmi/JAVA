		SELECT
	2. DML (Data Manipulation Language)
		INSERT
		UPDATE
		DELETE
	3. DDL (Date Definition Language)
		CREATE
		ALTER
		DROP
		TRUNCATE
		RENAME

		GRANT 권한부여
		REVOKE 권한해제

		COMMIT
		ROLLBACK
